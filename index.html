Connect in Seconds

Standard connection strings. Works with every ORM and driver you already use.

# Connect to your RoadDB PostgreSQL instance
import psycopg2

conn = psycopg2.connect(
    "postgresql://user:pass@your-db.roaddb.blackroad.io:5432/myapp"
)

# Or use MongoDB
from pymongo import MongoClient

client = MongoClient(
    "mongodb+srv://user:pass@your-db.roaddb.blackroad.io/myapp"
)

# Or Redis
import redis

r = redis.Redis.from_url(
    "rediss://user:pass@your-db.roaddb.blackroad.io:6379"
)
